#### 防抖

```
debounce (fn,wait) {
    if (this.fun!==null){
        clearTimeout(this.fun)
    }
    this.fun = setTimeout(fn,wait)
}
handleChangeCustomerId (val) {
    console.log('val =', val)
    this.$emit("change", val);
    if(val.trim().length !== 0) {
        this.debounce(this.checkCustomerId, 500);//500ms
    }
    if(val === '') {// 清空
        this.debounce(this.checkCustomerId, 500)
    }
}
```
